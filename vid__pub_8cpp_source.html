50 cv::Mat rotate(cv::Mat src, double angle)
51 {
52  // get rotation matrix for rotating the image around its center
53  cv::Point2f center(src.cols / 2.0, src.rows / 2.0);
54  cv::Mat rot = cv::getRotationMatrix2D(center, angle, 1.0);
55  // determine bounding rectangle
56  cv::Rect bbox = cv::RotatedRect(center, src.size(), angle).boundingRect();
57  // adjust transformation matrix
58  rot.at<double>(0, 2) += bbox.width / 2.0 - center.x;
59  rot.at<double>(1, 2) += bbox.height / 2.0 - center.y;
60  cv::Mat dst;
61  cv::warpAffine(src, dst, rot, bbox.size());
62  return dst;
63 }
