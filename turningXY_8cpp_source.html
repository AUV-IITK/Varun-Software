62  // called when new goal recieved; start motion and finish it, if not
63  // interupted
64  void analysisCB(const motion_commons::TurnGoalConstPtr goal)
65  {
66  ROS_INFO("%s Inside analysisCB, Goal = %f", ros::this_node::getName().c_str(), goal->AngleToTurn);
67 
68  int count = 0;
69  int loopRate = 10;
70  ros::Rate loop_rate(loopRate);
71 
72  // waiting till we recieve the first value from IMU else it's useless to any
73  // calculations
74  while (!initData)
75  {
76  ROS_INFO("%s Waiting to get first input from IMU", ros::this_node::getName().c_str());
77  loop_rate.sleep();
78  ros::spinOnce();
79  }
80 
81  finalAngularPosition = presentAngularPosition + goal->AngleToTurn;
82  if (finalAngularPosition >= 180)
83  finalAngularPosition = finalAngularPosition - 360;
84  else if (finalAngularPosition <= -180)
85  finalAngularPosition = finalAngularPosition + 360;
86 
87  float derivative = 0, integral = 0, dt = 1.0 / loopRate;
88  bool reached = false;
89  pwm.data = 0;
90 
91  if (!turnServer_.isActive())
92  return;
93 
94  while (!turnServer_.isPreemptRequested() && ros::ok() && count < goal->loop)
95  {
96  if (goal->loop > 10000)
97  {
98  p = p_stablize;
99  i = i_stablize;
100  d = d_stablize;
101  band = band_stablize;
102  }
103 
104  else
105  {
106  p = p_turn;
107  i = i_turn;
108  d = d_turn;
109  band = band_turn;
110  }
111 
112  error = finalAngularPosition - presentAngularPosition;
113  if (error < 0)
114  error_val = error + 360;
115  else
116  error_val = error - 360;
117 
118  if (abs(error_val) < abs(error))
119  error = error_val;
120 
121  integral += (error * dt);
122  derivative = (presentAngularPosition - previousAngularPosition) / dt;
123  output = (p * error) + (i * integral) + (d * derivative);
124  turningOutputPWMMapping(output);
125 
126  if (error < band && error > -band)
127  {
128  reached = true;
129  pwm.data = 0;
130  PWM.publish(pwm);
131  ROS_INFO("%s thrusters stopped", ros::this_node::getName().c_str());
132  count++;
133  }
134  else
135  {
136  reached = false;
137  count = 0;
138  }
139 
140  if (turnServer_.isPreemptRequested() || !ros::ok())
141  {
142  ROS_INFO("%s: Preempted", action_name_.c_str());
143  // set the action state to preempted
144  turnServer_.setPreempted();
145  reached = false;
146  break;
147  }
148 
149  feedback_.AngleRemaining = error;
150  turnServer_.publishFeedback(feedback_);
151  PWM.publish(pwm);
152  ROS_INFO("%s pwm send to arduino turn %d", ros::this_node::getName().c_str(), pwm.data);
153 
154  ros::spinOnce();
155  loop_rate.sleep();
156  }
157  if (reached)
158  {
159  result_.Result = reached;
160  ROS_INFO("%s: Succeeded", action_name_.c_str());
161  // set the action state to succeeded
162  turnServer_.setSucceeded(result_);
163  }
164  }
165  void turningOutputPWMMapping(float output)
166  {
167  float maxOutput = 1000, minOutput = -maxOutput;
168  float scale = 255 / maxOutput;
169  if (output > maxOutput)
170  output = maxOutput;
171  if (output < minOutput)
172  output = minOutput;
173  float temp = output * scale;
174  int output_pwm = static_cast<int>(temp);
175  if (output_pwm > 255)
176  output_pwm = 255;
177  if (output_pwm < -255)
178  output_pwm = -255;
179  pwm.data = output_pwm;
180  }
