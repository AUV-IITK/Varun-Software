146  void analysisCB(const task_commons::buoyGoalConstPtr goal)
147  {
148  ROS_INFO("Inside analysisCB");
149  sideCenter = false;
150  successBuoy = true;
151  IP_stopped = false;
152  heightGoal = false;
153  min_pwm = 4;
154  ros::Rate looprate(12);
155  if (!buoy_server_.isActive())
156  return;
157 
158  ROS_INFO("%s Waiting for Forward server to start.", action_name_.c_str());
159  ForwardClient_.waitForServer();
160  SidewardClient_.waitForServer();
161  UpwardClient_.waitForServer();
162  TurnClient_.waitForServer();
163 
164  TaskBuoyInnerClass::startBuoyDetection();
165 
166 
167  // Stabilization of yaw
168  turngoal.AngleToTurn = 0;
169  turngoal.loop = 100000;
170  TurnClient_.sendGoal(turngoal);
171  sleep(1);
172  if (present_Y_coord < 5.0 && present_Y_coord > -5.0)
173  {
174  ROS_INFO("y coordinate withing ip error range");
175  upwardgoal.Goal = present_depth;
176  }
177 
178  while (present_Y_coord > 5.0 || present_Y_coord < -5.0)
179  {
180  if (buoy_server_.isPreemptRequested() || !ros::ok())
181  {
182  ROS_INFO("%s: Preempted", action_name_.c_str());
183  // set the action state to preempted
184  buoy_server_.setPreempted();
185  successBuoy = false;
186  break;
187  }
188 
189  if (present_Y_coord < 5.0 && present_Y_coord > -5.0)
190  {
191  ROS_INFO("y coordinate withing ip error range");
192  upwardgoal.Goal = present_depth;
193  break;
194  }
195  else if (present_depth < 0.0)
196  {
197  min_pwm_data.data = min_pwm;
198  upward_pwm.publish(min_pwm_data);
199  }
200  else
201  {
202  min_pwm_data.data = -min_pwm;
203  upward_pwm.publish(min_pwm_data);
204  }
205  looprate.sleep();
206  feedback_.x_coord = data_X_.data;
207  feedback_.y_coord = present_Y_coord;
208  feedback_.distance = data_distance_.data;
209  buoy_server_.publishFeedback(feedback_);
210  ros::spinOnce();
211  }
212 
213  upwardgoal.loop = 100000;
214  UpwardClient_.sendGoal(upwardgoal);
215  ROS_INFO("%s: upward Stabilization is on", action_name_.c_str());
216 
217  sidewardgoal.Goal = 0;
218  sidewardgoal.loop = 10;
219  SidewardClient_.sendGoal(sidewardgoal);
220  boost::thread spin_thread_sideward_camera(&TaskBuoyInnerClass::spinThreadSidewardCamera, this);
221 
222  while (goal->order && successBuoy)
223  {
224  if (buoy_server_.isPreemptRequested() || !ros::ok())
225  {
226  ROS_INFO("%s: Preempted", action_name_.c_str());
227  // set the action state to preempted
228  buoy_server_.setPreempted();
229  successBuoy = false;
230  break;
231  }
232  looprate.sleep();
233  if (sideCenter)
234  {
235  break;
236  }
237  // publish the feedback
238  feedback_.x_coord = data_X_.data;
239  feedback_.y_coord = present_Y_coord;
240  feedback_.distance = data_distance_.data;
241  buoy_server_.publishFeedback(feedback_);
242  ros::spinOnce();
243  }
244 
245  ROS_INFO("%s: Bot is in center of buoy", action_name_.c_str());
246  forwardgoal.Goal = 0;
247  forwardgoal.loop = 10;
248  ForwardClient_.sendGoal(forwardgoal);
249  ROS_INFO("%s: Bot is moving forward to hit the buoy", action_name_.c_str());
250 
251  while (goal->order && successBuoy)
252  {
253  if (buoy_server_.isPreemptRequested() || !ros::ok())
254  {
255  ROS_INFO("%s: Preempted", action_name_.c_str());
256  // set the action state to preempted
257  buoy_server_.setPreempted();
258  successBuoy = false;
259  break;
260  }
261  looprate.sleep();
262 
263  if (IP_stopped)
264  {
265  ROS_INFO("%s: Waiting for hitting the buoy...", action_name_.c_str());
266  // sleep(1);
267  break;
268  }
269 
270  feedback_.x_coord = data_X_.data;
271  feedback_.y_coord = present_Y_coord;
272  feedback_.distance = data_distance_.data;
273  buoy_server_.publishFeedback(feedback_);
274  ros::spinOnce();
275  }
276 
277  ForwardClient_.cancelGoal(); // stop motion here
278 
279  result_.MotionCompleted = successBuoy;
280  ROS_INFO("%s: Succeeded", action_name_.c_str());
281  // set the action state to succeeded
282  buoy_server_.setSucceeded(result_);
283  }
