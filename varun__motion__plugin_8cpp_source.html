52  void PWMCbForward(const std_msgs::Int32& msg)
53  {
54  math::Vector3 force;
55  math::Vector3 reset_force;
56  int pwm = msg.data;
57  force.x = FULL_FORCE * pwm / 255 * ERROR_FACTOR;
58  _model->GetLink("body")->SetForce(reset_force);
59  _model->GetLink("body")->AddRelativeForce(force);
60  }
61 
62  void PWMCbSideward(const std_msgs::Int32& msg)
63  {
64  math::Vector3 force;
65  math::Vector3 reset_force;
66  int pwm = msg.data;
67  pwm = -pwm; // this is because motionlibrary assumes right to be positive while gazebo y axis points to left.
68  force.y = FULL_FORCE * pwm / 255 * ERROR_FACTOR;
69  _model->GetLink("body")->SetForce(reset_force);
70  _model->GetLink("body")->AddRelativeForce(force);
71  }
72 
73  void PWMCbUpward(const std_msgs::Int32& msg)
74  {
75  math::Vector3 force;
76  int pwm = msg.data;
77  force.z = FULL_FORCE * pwm / 255 * ERROR_FACTOR;
78  _model->GetLink("body")->SetForce(force);
79  }
80 
81  void PWMCbTurn(const std_msgs::Int32& msg)
82  {
83  math::Vector3 torque;
84  int pwm = msg.data;
85  torque.z = FULL_FORCE * pwm / 255 * (VARUN_SIDEWARD_LENGTH / 2) * ERROR_FACTOR;
86  _model->GetLink("body")->SetTorque(torque);
87  }
