63  // called when new goal recieved; Start motion and finish it, if not interrupted
64  void analysisCB(const motion_commons::UpwardGoalConstPtr goal)
65  {
66  ROS_INFO("%s: Inside analysisCB", ros::this_node::getName().c_str());
67 
68  int count = 0;
69  int loopRate = 10;
70  ros::Rate loop_rate(loopRate);
71 
72  // waiting till we recieve the first value from Camera/pressure sensor else it's useless do any calculations
73  while (!initData)
74  {
75  ROS_INFO("%s: Waiting to get first input at topic zDistance", ros::this_node::getName().c_str());
76  loop_rate.sleep();
77  }
78 
79  finalDepth = goal->Goal;
80 
81  float derivative = 0, integral = 0, dt = 1.0 / loopRate;
82  bool reached = false;
83  pwm.data = 0;
84 
85  if (!upwardServer_.isActive())
86  return;
87 
88  while (!upwardServer_.isPreemptRequested() && ros::ok() && count < goal->loop)
89  {
90  if (goal->loop > 10000)
91  {
92  p = p_stablize;
93  i = i_stablize;
94  d = d_stablize;
95  band = band_stablize;
96  }
97 
98  else
99  {
100  p = p_upward;
101  i = i_upward;
102  d = d_upward;
103  band = band_upward;
104  }
105 
106  error = finalDepth - presentDepth;
107  integral += (error * dt);
108  derivative = (presentDepth - previousDepth) / dt;
109  output = (p * error) + (i * integral) + (d * derivative);
110  upwardOutputPWMMapping(output);
111 
112  if (pwm.data <= band && pwm.data >= -band)
113  {
114  reached = true;
115  pwm.data = 0;
116  PWM.publish(pwm);
117  ROS_INFO("%s: thrusters stopped", ros::this_node::getName().c_str());
118  count++;
119  }
120  else
121  {
122  reached = false;
123  count = 0;
124  }
125 
126  if (upwardServer_.isPreemptRequested() || !ros::ok())
127  {
128  ROS_INFO("%s: %s: Preempted", ros::this_node::getName().c_str(), action_name_.c_str());
129  // set the action state to preempted
130  upwardServer_.setPreempted();
131  reached = false;
132  break;
133  }
134 
135  feedback_.DepthRemaining = error;
136  upwardServer_.publishFeedback(feedback_);
137  PWM.publish(pwm);
138  ROS_INFO("%s: pwm send to arduino upward %d", ros::this_node::getName().c_str(), pwm.data);
139 
140  ros::spinOnce();
141  loop_rate.sleep();
142  }
143  if (reached)
144  {
145  result_.Result = reached;
146  ROS_INFO("%s: %s: Succeeded", ros::this_node::getName().c_str(), action_name_.c_str());
147  // set the action state to succeeded
148  upwardServer_.setSucceeded(result_);
149  }
150  }

152  void upwardOutputPWMMapping(float output)
153  {
154  const float maxOutput = 1000, minOutput = -maxOutput;
155  const float scale = 255 / maxOutput;
156  if (output > maxOutput)
157  output = maxOutput;
158  if (output < minOutput)
159  output = minOutput;
160  float temp = output * scale;
161  int output_pwm = static_cast<int>(temp);
162  if (output_pwm > maxPwm)
163  output_pwm = maxPwm;
164  if (output_pwm < -maxPwm)
165  output_pwm = -maxPwm;
166  pwm.data = output_pwm;
167  }
