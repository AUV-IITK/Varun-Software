173  void analysisCB(const task_commons::lineGoalConstPtr goal)
174  {
175  ROS_INFO("Inside analysisCB");
176  success = true;
177  isOrange = false;
178  LineAlign = false;
179  FrontCenter = false;
180  SideCenter = false;
181  ros::Rate looprate(12);
182 
183  if (!line_server_.isActive())
184  return;
185 
186  ROS_INFO("%s Waiting for Forward server to start.", action_name_.c_str());
187  ForwardClient_.waitForServer();
188  SidewardClient_.waitForServer();
189  TurnClient_.waitForServer();
190 
191  TaskLineInnerClass::detection_switch_on();
192 
193  // Stabilization of yaw
194  turngoal.AngleToTurn = 0;
195  turngoal.loop = 100000;
196  TurnClient_.sendGoal(turngoal);
197  // start moving forward.
198  // send initial data to forward.
199  forwardgoal.Goal = 100;
200  forwardgoal.loop = 10;
201  ForwardClient_.sendGoal(forwardgoal);
202  ROS_INFO("%s: searching line", action_name_.c_str());
203 
204  while (goal->order && success)
205  {
206  if (line_server_.isPreemptRequested() || !ros::ok())
207  {
208  ROS_INFO("%s: Preempted", action_name_.c_str());
209  // set the action state to preempted
210  line_server_.setPreempted();
211  success = false;
212  break;
213  }
214  looprate.sleep();
215 
216  if (isOrange)
217  {
218  // stop will happen outside while loop so if preempted then too it will
219  // stop
220  break;
221  }
222  // publish the feedback
223 
224  feedback_.AngleRemaining = angle_goal.data;
225  feedback_.x_coord = data_X_.data;
226  feedback_.y_coord = data_Y_.data;
227  line_server_.publishFeedback(feedback_);
228  ros::spinOnce();
229  }
230 
231  ForwardClient_.cancelGoal(); // stop motion here
232  TaskLineInnerClass::detection_switch_off();
233 
234  TaskLineInnerClass::centralize_switch_on();
235 
236  sidewardgoal.Goal = 0;
237  sidewardgoal.loop = 10;
238  SidewardClient_.sendGoal(sidewardgoal);
239  boost::thread spin_thread_sideward_camera(&TaskLineInnerClass::spinThreadSidewardCamera, this);
240 
241  forwardgoal.Goal = 0;
242  forwardgoal.loop = 10;
243  ForwardClient_.sendGoal(forwardgoal);
244  boost::thread spin_thread_forward_camera(&TaskLineInnerClass::spinThreadForwardCamera, this);
245  ROS_INFO("%s: line is going to be centralized", action_name_.c_str());
246 
247  while (goal->order && success)
248  {
249  if (line_server_.isPreemptRequested() || !ros::ok())
250  {
251  ROS_INFO("%s: Preempted", action_name_.c_str());
252  // set the action state to preempted
253  line_server_.setPreempted();
254  success = false;
255  break;
256  }
257  looprate.sleep();
258  if (FrontCenter && SideCenter)
259  {
260  ROS_INFO("%s: Line has been centralized.", action_name_.c_str());
261  break;
262  }
263  // publish the feedback
264  feedback_.AngleRemaining = angle_goal.data;
265  feedback_.x_coord = data_X_.data;
266  feedback_.y_coord = data_Y_.data;
267  line_server_.publishFeedback(feedback_);
268  ros::spinOnce();
269  }
270 
271  TaskLineInnerClass::centralize_switch_off();
272 
273  TaskLineInnerClass::angle_switch_on();
274 
275  TurnClient_.cancelGoal(); // stopped stablisation
276 
277  while (angle_goal.data == 0.0)
278  {
279  sleep(.01);
280  }
281  turngoal.AngleToTurn = angle_goal.data;
282  turngoal.loop = 10;
283  TurnClient_.sendGoal(turngoal);
284  boost::thread spin_thread_turn_camera(&TaskLineInnerClass::spinThreadTurnCamera, this);
285  ROS_INFO("%s: line is going to be aligned", action_name_.c_str());
286 
287  while (goal->order && success)
288  {
289  if (line_server_.isPreemptRequested() || !ros::ok())
290  {
291  ROS_INFO("%s: Preempted", action_name_.c_str());
292  // set the action state to preempted
293  line_server_.setPreempted();
294  success = false;
295  break;
296  }
297 
298  looprate.sleep();
299  if (LineAlign)
300  {
301  ROS_INFO("%s: line is aligned.", action_name_.c_str());
302  break;
303  }
304 
305  feedback_.AngleRemaining = angle_goal.data;
306  feedback_.x_coord = data_X_.data;
307  feedback_.y_coord = data_Y_.data;
308  line_server_.publishFeedback(feedback_);
309  ros::spinOnce();
310  }
311 
312  TaskLineInnerClass::angle_switch_off();
313 
314  result_.MotionCompleted = success;
315  ROS_INFO("%s: Success is %s", action_name_.c_str(), success ? "true" : "false");
316  line_server_.setSucceeded(result_);
317  }
